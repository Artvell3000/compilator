ARRAY a(4); a[0]:=4; a[1]:=3; a[2]:=2; a[3]:=1;
VAR n:=4;
VAR i:=0;
VAR j:=0;
VAR t:=0;
WHILE( i < n ) DO {
    j:=0;
    WHILE( j < n - 1 ) DO {
        IF ( a[j] > a[j+1] ) THEN {
            t:=a[j];
            a[j]:=a[j+1];
            a[j+1]:=t;
        };
        j:=j+1;
    };
    i:=i+1;
};
OUTPUT a[0];
OUTPUT a[1];
OUTPUT a[2];
OUTPUT a[3];
